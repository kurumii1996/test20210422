create database test;

 

drop table usr;

 

create table usr (
              mailaddr varchar(255) not null,
              user_id varchar(16) ,
              password varchar(16) not null,
              user_name varchar(64) not null,
              user_auth varchar(1) not null,
              primary key(mailaddr, user_id)
);

 

create table vacation (
     adres varchar(255) primary key ,
     ymd varchar(8),
     type varchar(1),
     apply_f varchar(1),
     reason varchar(1024)
);

 

select * from usr;

 

select user_name, ymd, type, apply_f, reason 
from usr 
inner join vacation
on usr.mailaddr = vacation.adres;

 

select user_name, ymd, type, apply_f, reason 
from usr 
left join vacation
on usr.mailaddr = vacation.adres;

 

select user_name, ymd, type, apply_f, reason 
from usr 
right join vacation
on usr.mailaddr = vacation.adres;

 

select user_name, vacation.ymd, vacation.type, vacation.apply_f, vacation.reason 
from usr 
full outer join vacation
on usr.mailaddr = vacation.adres;
